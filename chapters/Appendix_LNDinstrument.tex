function Ibias() {
    a = $(HK_Ibias+3)
    if (a>4000) a = (147.3 + degC($(HK_T_LVPS+3))*0.164 - $(HK_Vbias+3)*0.05488) / 0.01810
    return a*0.4928
}

function Vbias() {
    return $(HK_Vbias+3)*0.05488 + $(HK_Ibias+3)*0.01810
}
